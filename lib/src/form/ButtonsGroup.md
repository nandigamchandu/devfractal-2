### Default ButtonsGroup

```jsx
import { Button } from './Button'
;<ButtonsGroup>
  <Button>Submit</Button>
  <Button>Reset</Button>
</ButtonsGroup>
```

### ButtonsGroup with all properties

```jsx
import { Button } from './Button'
;<ButtonsGroup addons alignment="centered">
  <Button>Submit</Button>
  <Button>Reset</Button>
</ButtonsGroup>
```
